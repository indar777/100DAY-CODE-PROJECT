/*
 write a Program to print the following pattern:
  1
  12
  123
  1234
  12345

  For n = 5, each row prints numbers starting from 1 up to the current row number.
*/

#include <stdio.h>

int main() {
    int i, j, rows;

    
    printf("Enter number of rows: ");
    scanf("%d", &rows);

    
    for (i = 1; i <= rows; i++) {
        // Inner loop prints numbers from 1 to the current row number
        for (j = 1; j <= i; j++) {
            printf("%d", j);
        }
        
        printf("\n");
    }

    return 0;
}
